Módulo Inicial

Construindo a API

Vamos construir o coração do seu sistema. Uma API Restful capaz de registrar usuários, gerar tokens de segurança (JWT), salvar tarefas financeiras e devolver relatórios calculados no banco de dados.

Autenticação JWT

A API não tem "sessão" ativa. O usuário loga, recebe um token criptografado e deve enviar esse token no cabeçalho para acessar as tarefas dele.

Agrupamento SQL

Vamos fazer o cálculo de tarefas e soma de valores direto no banco de dados usando SQLAlchemy (muito mais rápido que somar no Python).

2. Configuração e Modelos (app.py)

from flask import Flask, request, jsonify
from flask_sqlalchemy import SQLAlchemy
from flask_jwt_extended import JWTManager, create_access_token, jwt_required, get_jwt_identity
from werkzeug.security import generate_password_hash, check_password_hash
from datetime import datetime

app = Flask(__name__)
app.config['SQLALCHEMY_DATABASE_URI'] = 'sqlite:///banco.db'
app.config['JWT_SECRET_KEY'] = 'super-chave-secreta-mudar-em-prod'

db = SQLAlchemy(app)
jwt = JWTManager(app)

class User(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    username = db.Column(db.String(80), unique=True, nullable=False)
    password_hash = db.Column(db.String(256), nullable=False)
    tasks = db.relationship('Task', backref='owner', lazy=True)

class Task(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    title = db.Column(db.String(100), nullable=False)
    description = db.Column(db.Text)
    date = db.Column(db.Date, nullable=False)
    value = db.Column(db.Float, default=0.0)
    user_id = db.Column(db.Integer, db.ForeignKey('user.id'), nullable=False)

with app.app_context():
    db.create_all()
PASSO 2

Cadastro e Login

Criando Usuário

@app.route('/register', methods=['POST'])
def register():
    data = request.get_json()
    if User.query.filter_by(username=data['username']).first():
        return jsonify({"error": "Usuário já existe"}), 400

    hashed_password = generate_password_hash(data['password'])
    new_user = User(username=data['username'], password_hash=hashed_password)
    db.session.add(new_user)
    db.session.commit()
    
    return jsonify({"message": "Usuário criado com sucesso!"}), 201

Efetuando Login

@app.route('/login', methods=['POST'])
def login():
    data = request.get_json()
    user = User.query.filter_by(username=data['username']).first()

    if not user or not check_password_hash(user.password_hash, data['password']):
        return jsonify({"error": "Credenciais inválidas"}), 401

    access_token = create_access_token(identity=str(user.id))
    return jsonify({"token": access_token}), 200
PASSO 3

Criando Tarefas

@app.route('/tasks', methods=['POST'])
@jwt_required()
def create_task():
    current_user_id = get_jwt_identity()
    data = request.get_json()

    task_date = datetime.strptime(data['date'], '%Y-%m-%d').date()

    new_task = Task(
        title=data['title'],
        description=data.get('description', ""),
        date=task_date,
        value=data.get('value', 0.0),
        user_id=current_user_id
    )

    db.session.add(new_task)
    db.session.commit()
    return jsonify({"message": "Tarefa registrada!"}), 201

Agrupamento SQL

from sqlalchemy import func

@app.route('/tasks/summary', methods=['GET'])
@jwt_required()
def get_tasks_summary():
    current_user_id = get_jwt_identity()
    
    daily_stats = db.session.query(
        Task.date,
        func.count(Task.id).label('total_tasks'),
        func.sum(Task.value).label('total_value')
    ).filter(Task.user_id == current_user_id) \
     .group_by(Task.date).all()

    daily_result = [
        {
            "date": stat.date.strftime('%Y-%m-%d'),
            "tasks_count": stat.total_tasks,
            "total_value": stat.total_value or 0.0
        } for stat in daily_stats
    ]

    return jsonify({"daily_summary": daily_result}), 200

if __name__ == '__main__':
    app.run(debug=True, port=5000)
